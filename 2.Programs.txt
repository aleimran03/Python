#Program for a concession stand

menu = {"pizza": 3.00,
        "nachos": 4.00,
        "popcorn": 6.00,
        "fries": 2.34,
        "chips": 2.22,
        "pretzel": 1.47,
        "soda": 3.94,
        "lemonide": 2.99}

cart = []
total = 0

for key, value in menu.items():
    print(f"{key:10} : ${value:.2F}")

while True:
    food = input("Select an item (Press q to Quit) :").lower()
    if food == "q":
      break
    elif menu.get(food) is not None:
      cart.append(food)

print("--------YOUR ORDER IS--------")

for food in cart:
    total += menu.get(food)
    print(food, end=" ")
print()
print(f"Total Price is : ${total:.2f}")

#List Comprehension = A concise way to create lists in python
			compact and easier to read than traditional loops
			[expression for value in iterable if condition]

double = []

for x in range(1, 11):
	double.append(x * 2)
print(double)

#we can write in more compact way 

#double [expression for value in iterable]
doubles = [x * 2 for x in range(1, 11)]
triple = [y * 3 for y in range(1, 11)]
square = [z * z for z in range(1, 11)]

print(f"Double is {doubles}")
print(f"Triple is {triple}")
print(f"Square is {square}")

Match-case statement (switch): An alternative to using many 'elif' statements
				Execute some code if a value matches a 'case'
				Benefits: cleaner and syntax is more readable


def day_of_the_week(day):
  match day:
    case 0:
      return "Its a Sunday"
    case 1:
      return "Its a Monday"
    case 3:
      return "Its a Wednesday"
    case 4:
      return "Its a Thursday"
    case 5:
      return "Its a Friday"
    case 6:
      return "Its a saturday"
    case _:
      return "Invalid day"
print(day_of_the_week(2))

#We can use and or boolean operator in match statements

def is_weekend(day):
  match day:
    case "Monday" | "Tuesday" | "Wednesday" | "Thursday" | "Friday":
      return False
    case "Saturday" | "Sunday":
      return True
    case _:
      return "Invalid day"
print(is_weekend("Saturday"))
